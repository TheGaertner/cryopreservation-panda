87 bool MotionGenerator::calculateDesiredValues(double t, Vector7d* delta_q_d) const {
88  Vector7i sign_delta_q;
89  sign_delta_q << delta_q_.cwiseSign().cast<int>();
90  Vector7d t_d = t_2_sync_ - t_1_sync_;
91  Vector7d delta_t_2_sync = t_f_sync_ - t_2_sync_;
92  std::array<bool, 7> joint_motion_finished{};
93 
94  for (size_t i = 0; i < 7; i++) {
95  if (std::abs(delta_q_[i]) < kDeltaQMotionFinished) {
96  (*delta_q_d)[i] = 0;
97  joint_motion_finished[i] = true;
98  } else {
99  if (t < t_1_sync_[i]) {
100  (*delta_q_d)[i] = -1.0 / std::pow(t_1_sync_[i], 3.0) * dq_max_sync_[i] * sign_delta_q[i] *
101  (0.5 * t - t_1_sync_[i]) * std::pow(t, 3.0);
102  } else if (t >= t_1_sync_[i] && t < t_2_sync_[i]) {
103  (*delta_q_d)[i] = q_1_[i] + (t - t_1_sync_[i]) * dq_max_sync_[i] * sign_delta_q[i];
104  } else if (t >= t_2_sync_[i] && t < t_f_sync_[i]) {
105  (*delta_q_d)[i] =
106  delta_q_[i] + 0.5 *
107  (1.0 / std::pow(delta_t_2_sync[i], 3.0) *
108  (t - t_1_sync_[i] - 2.0 * delta_t_2_sync[i] - t_d[i]) *
109  std::pow((t - t_1_sync_[i] - t_d[i]), 3.0) +
110  (2.0 * t - 2.0 * t_1_sync_[i] - delta_t_2_sync[i] - 2.0 * t_d[i])) *
111  dq_max_sync_[i] * sign_delta_q[i];
112  } else {
113  (*delta_q_d)[i] = delta_q_[i];
114  joint_motion_finished[i] = true;
115  }
116  }
117  }
118  return std::all_of(joint_motion_finished.cbegin(), joint_motion_finished.cend(),
119  [](bool x) { return x; });
120 }

122 void MotionGenerator::calculateSynchronizedValues() {
123  Vector7d dq_max_reach(dq_max_);
124  Vector7d t_f = Vector7d::Zero();
125  Vector7d delta_t_2 = Vector7d::Zero();
126  Vector7d t_1 = Vector7d::Zero();
127  Vector7d delta_t_2_sync = Vector7d::Zero();
128  Vector7i sign_delta_q;
129  sign_delta_q << delta_q_.cwiseSign().cast<int>();
130 
131  for (size_t i = 0; i < 7; i++) {
132  if (std::abs(delta_q_[i]) > kDeltaQMotionFinished) {
133  if (std::abs(delta_q_[i]) < (3.0 / 4.0 * (std::pow(dq_max_[i], 2.0) / ddq_max_start_[i]) +
134  3.0 / 4.0 * (std::pow(dq_max_[i], 2.0) / ddq_max_goal_[i]))) {
135  dq_max_reach[i] = std::sqrt(4.0 / 3.0 * delta_q_[i] * sign_delta_q[i] *
136  (ddq_max_start_[i] * ddq_max_goal_[i]) /
137  (ddq_max_start_[i] + ddq_max_goal_[i]));
138  }
139  t_1[i] = 1.5 * dq_max_reach[i] / ddq_max_start_[i];
140  delta_t_2[i] = 1.5 * dq_max_reach[i] / ddq_max_goal_[i];
141  t_f[i] = t_1[i] / 2.0 + delta_t_2[i] / 2.0 + std::abs(delta_q_[i]) / dq_max_reach[i];
142  }
143  }
144  double max_t_f = t_f.maxCoeff();
145  for (size_t i = 0; i < 7; i++) {
146  if (std::abs(delta_q_[i]) > kDeltaQMotionFinished) {
147  double a = 1.5 / 2.0 * (ddq_max_goal_[i] + ddq_max_start_[i]);
148  double b = -1.0 * max_t_f * ddq_max_goal_[i] * ddq_max_start_[i];
149  double c = std::abs(delta_q_[i]) * ddq_max_goal_[i] * ddq_max_start_[i];
150  double delta = b * b - 4.0 * a * c;
151  if (delta < 0.0) {
152  delta = 0.0;
153  }
154  dq_max_sync_[i] = (-1.0 * b - std::sqrt(delta)) / (2.0 * a);
155  t_1_sync_[i] = 1.5 * dq_max_sync_[i] / ddq_max_start_[i];
156  delta_t_2_sync[i] = 1.5 * dq_max_sync_[i] / ddq_max_goal_[i];
157  t_f_sync_[i] =
158  (t_1_sync_)[i] / 2.0 + delta_t_2_sync[i] / 2.0 + std::abs(delta_q_[i] / dq_max_sync_[i]);
159  t_2_sync_[i] = (t_f_sync_)[i] - delta_t_2_sync[i];
160  q_1_[i] = (dq_max_sync_)[i] * sign_delta_q[i] * (0.5 * (t_1_sync_)[i]);
161  }
162  }
163 }

165 franka::JointPositions MotionGenerator::operator()(const franka::RobotState& robot_state,
166  franka::Duration period) {
167  time_ += period.toSec();
168 
169  if (time_ == 0.0) {
170  q_start_ = Vector7d(robot_state.q_d.data());
171  delta_q_ = q_goal_ - q_start_;
172  calculateSynchronizedValues();
173  }
174 
175  Vector7d delta_q_d;
176  bool motion_finished = calculateDesiredValues(time_, &delta_q_d);
177 
178  std::array<double, 7> joint_positions;
179  Eigen::VectorXd::Map(&joint_positions[0], 7) = (q_start_ + delta_q_d);
180  franka::JointPositions output(joint_positions);
181  output.motion_finished = motion_finished;
182  return output;
